dbr <- subset(dbr, gs_name %in% use)

Build gene expression rankings for each cell:

rankings <- AUCell_buildRankings(
  exprMat = counts(sce), 
  plotStats = FALSE, 
  verbose = FALSE
)

Calculate the AUC for each gene-set in each cell:
